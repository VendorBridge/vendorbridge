CREATE SCHEMA "public";
CREATE TYPE "user_role" AS ENUM('ADMIN', 'PROCUREMENT_OFFICER', 'VENDOR', 'MANAGER');
CREATE TYPE "user_status" AS ENUM('ACTIVE', 'INACTIVE', 'SUSPENDED');
CREATE TYPE "vendor_status" AS ENUM('ACTIVE', 'INACTIVE', 'PENDING_VERIFICATION', 'BLACKLISTED');
CREATE TYPE "vendor_category" AS ENUM('IT_HARDWARE', 'IT_SOFTWARE', 'FURNITURE', 'STATIONERY', 'LOGISTICS', 'FACILITY', 'MARKETING', 'FINANCE', 'OTHER');
CREATE TYPE "rfq_status" AS ENUM('DRAFT', 'PUBLISHED', 'CLOSED', 'CANCELLED', 'AWARDED');
CREATE TYPE "quotation_status" AS ENUM('DRAFT', 'SUBMITTED', 'UNDER_REVIEW', 'SHORTLISTED', 'REJECTED', 'ACCEPTED');
CREATE TYPE "approval_status" AS ENUM('PENDING', 'APPROVED', 'REJECTED', 'ESCALATED');
CREATE TYPE "po_status" AS ENUM('DRAFT', 'ISSUED', 'ACKNOWLEDGED', 'PARTIALLY_FULFILLED', 'FULFILLED', 'CANCELLED');
CREATE TYPE "invoice_status" AS ENUM('DRAFT', 'ISSUED', 'SENT', 'PAID', 'OVERDUE', 'CANCELLED');
CREATE TYPE "activity_entity" AS ENUM('USER', 'VENDOR', 'RFQ', 'QUOTATION', 'APPROVAL', 'PURCHASE_ORDER', 'INVOICE');
CREATE TYPE "activity_action" AS ENUM('CREATED', 'UPDATED', 'DELETED', 'SUBMITTED', 'APPROVED', 'REJECTED', 'PUBLISHED', 'CLOSED', 'AWARDED', 'GENERATED', 'SENT', 'PAID', 'CANCELLED', 'SHORTLISTED', 'ACKNOWLEDGED', 'LOGIN', 'LOGOUT');
CREATE TYPE "notification_type" AS ENUM('RFQ_PUBLISHED', 'RFQ_CLOSED', 'QUOTATION_RECEIVED', 'QUOTATION_SHORTLISTED', 'QUOTATION_REJECTED', 'APPROVAL_REQUESTED', 'APPROVAL_GRANTED', 'APPROVAL_REJECTED', 'PO_ISSUED', 'PO_ACKNOWLEDGED', 'INVOICE_ISSUED', 'INVOICE_PAID', 'INVOICE_OVERDUE', 'SYSTEM');
CREATE TABLE "activity_logs" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"entity_type" activity_entity NOT NULL,
	"entity_id" uuid NOT NULL,
	"action" activity_action NOT NULL,
	"actor_id" uuid,
	"actor_name" varchar(200),
	"description" text NOT NULL,
	"old_value" jsonb,
	"new_value" jsonb,
	"metadata" jsonb,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "approvals" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"quotation_id" uuid NOT NULL,
	"rfq_id" uuid NOT NULL,
	"vendor_id" uuid NOT NULL,
	"status" approval_status DEFAULT 'PENDING' NOT NULL,
	"requested_by" uuid NOT NULL,
	"assigned_to" uuid,
	"decision_by" uuid,
	"decision_at" timestamp with time zone,
	"remarks" text,
	"timeline" jsonb DEFAULT '[]' NOT NULL,
	"priority" varchar(20) DEFAULT 'NORMAL' NOT NULL,
	"due_date" date,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "category_spend_stats" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"year" integer NOT NULL UNIQUE,
	"month" integer NOT NULL UNIQUE,
	"category" vendor_category NOT NULL UNIQUE,
	"total_spend" numeric(15, 2) DEFAULT '0.00' NOT NULL,
	"po_count" integer DEFAULT 0 NOT NULL,
	"computed_at" timestamp with time zone DEFAULT now() NOT NULL,
	CONSTRAINT "category_spend_stats_year_month_category_key" UNIQUE("year","month","category"),
	CONSTRAINT "category_spend_stats_month_check" CHECK (((month >= 1) AND (month <= 12)))
);
CREATE TABLE "invoice_items" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"invoice_id" uuid NOT NULL,
	"po_item_id" uuid,
	"item_name" varchar(300) NOT NULL,
	"description" text,
	"quantity" numeric(12, 3) DEFAULT '1' NOT NULL,
	"unit" varchar(50),
	"unit_price" numeric(12, 2) DEFAULT '0.00' NOT NULL,
	"tax_pct" numeric(5, 2) DEFAULT '18.00' NOT NULL,
	"line_subtotal" numeric(15, 2) GENERATED ALWAYS AS ((quantity * unit_price)) STORED,

);
CREATE TABLE "invoices" (
	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),

	"grand_total" numeric(15, 2) DEFAULT '0.00' NOT NULL,
	"payment_terms" varchar(200),
	"payment_method" varchar(100),
	"payment_reference" varchar(200),
	"billed_from_name" varchar(200),
	"billed_from_address" text,
	"billed_from_gstin" varchar(20),
	"billed_to_name" varchar(200),
	"billed_to_address" text,
	"billed_to_gstin" varchar(20),
	"pdf_url" text,
	"email_sent_to" text,
	"email_sent_at" timestamp with time zone,
	"printed_at" timestamp with time zone,
	"notes" text,
	"created_by" uuid NOT NULL,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);

	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),
	"year" integer NOT NULL UNIQUE,
	"month" integer NOT NULL UNIQUE,
	"total_rfqs" integer DEFAULT 0 NOT NULL,
	"total_quotations" integer DEFAULT 0 NOT NULL,

	"top_vendor_id" uuid,
	"top_category" vendor_category,

	"id" uuid PRIMARY KEY DEFAULT gen_random_uuid(),

	"po_id" uuid NOT NULL UNIQUE,
	"rated_by" uuid NOT NULL,
	"score" numeric(3, 2) NOT NULL,
	"quality" numeric(3, 2),
	"delivery" numeric(3, 2),
	"communication" numeric(3, 2),
	"comment" text,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	CONSTRAINT "vendor_ratings_vendor_id_po_id_key" UNIQUE("vendor_id","po_id"),
	CONSTRAINT "vendor_ratings_communication_check" CHECK (((communication >= (0)::numeric) AND (communication <= (5)::numeric))),
	CONSTRAINT "vendor_ratings_delivery_check" CHECK (((delivery >= (0)::numeric) AND (delivery <= (5)::numeric))),
	CONSTRAINT "vendor_ratings_quality_check" CHECK (((quality >= (0)::numeric) AND (quality <= (5)::numeric))),
	CONSTRAINT "vendor_ratings_score_check" CHECK (((score >= (0)::numeric) AND (score <= (5)::numeric)))
);
